twitter_dict = []

with open('tweet_json.txt', 'r') as openfile:
    try: 
        for line in openfile:
            tempdict = json.loads(line)
            twit_id = tempdict['id']
            retwit_count = tempdict['retweet_count']
            favorite_count = tempdict['favorite_count']
